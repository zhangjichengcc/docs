## Record<keys, Type>

interface object1Prop {
  [key: 'a'|'b']: string;
  // 索引签名参数类型不能为文本类型或泛型类型。请考虑改用映射的对象类型。ts(1337)

}

interface object2Prop {
  Record<'a'|'b', string>

}
